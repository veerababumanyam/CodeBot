interface Envelope<T> { status: string; data: T; }
interface PaginatedEnvelope<T> { status: string; data: T[]; pagination: {...} | null; }
export const agentApi = {
  list: (pipelineId: string) => api.get<PaginatedEnvelope<Agent>>(`/api/v1/agents?pipeline_id=${pipelineId}`),

};

**3. Create `src/lib/yjs-provider.ts`** -- Yjs WebSocket provider wrapper:
```typescript
import * as Y from "yjs";
import { WebsocketProvider } from "y-websocket";

export function createYjsProvider(docId: string, serverUrl?: string): { doc: Y.Doc; provider: WebsocketProvider } {
  const url = serverUrl ?? (import.meta.env.VITE_YJS_URL ?? "ws://localhost:1234");
  const doc = new Y.Doc();
  const provider = new WebsocketProvider(url, docId, doc, { connect: true });
  return { doc, provider };
}

export function destroyYjsProvider(provider: WebsocketProvider, doc: Y.Doc): void {
  provider.disconnect();
  provider.destroy();
  doc.destroy();
}
```

**4. Create `src/hooks/use-yjs.ts`** -- React hook for Yjs lifecycle:
