```python
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from enum import StrEnum
from typing import Any

class AgentState(StrEnum):
    IDLE = "idle"
    INITIALIZING = "initializing"
    EXECUTING = "executing"
    REVIEWING = "reviewing"
    COMPLETED = "completed"
    FAILED = "failed"
    RECOVERING = "recovering"

@dataclass(slots=True, kw_only=True)
class AgentResult:
    success: bool
    output: dict[str, Any] = field(default_factory=dict)
    error: str | None = None
    tokens_used: int = 0
    cost_usd: float = 0.0
    execution_time_ms: int = 0

class BaseAgent(ABC):
    def __init__(self, config: AgentConfig) -> None: ...
    @abstractmethod
    async def perceive(self, context: dict[str, Any]) -> dict[str, Any]: ...
    @abstractmethod
    async def reason(self, perception: dict[str, Any]) -> dict[str, Any]: ...
    @abstractmethod
    async def act(self, plan: dict[str, Any]) -> AgentResult: ...
    async def execute(self, context: dict[str, Any]) -> AgentResult: ...
    async def self_review(self, result: AgentResult) -> AgentResult: ...
```

```python
from __future__ import annotations
import uuid
from dataclasses import dataclass, field
from typing import Any
from codebot.agents.base import BaseAgent, AgentResult
from codebot.skills.models import Skill, SkillStatus
from codebot.skills.service import SkillService
from codebot.events.bus import EventBus

@dataclass(slots=True, kw_only=True)
class ExtractedPattern:
    name: str
    description: str
    parameterized_code: str
    applicable_agents: list[str]
    input_schema: dict[str, Any] = field(default_factory=dict)
    output_schema: dict[str, Any] = field(default_factory=dict)
    tags: list[str] = field(default_factory=list)

class SkillCreatorAgent(BaseAgent):
    """Agent #26: Extracts reusable code patterns into skills for other agents.

    Runs post-delivery in Stage S9. Analyzes completed project code to identify
    recurring patterns (auth flows, CRUD, form validation, API clients) and packages
    them as parameterized skills in the skill registry.
    """

    def __init__(self, config: Any, skill_service: SkillService, event_bus: EventBus) -> None:
        super().__init__(config)
        self._skill_service = skill_service
        self._event_bus = event_bus

    async def perceive(self, context: dict[str, Any]) -> dict[str, Any]:
        """Gather project source files and pipeline execution logs.

        Extracts: completed code files, execution patterns, agent outputs,
        and existing skill registry contents to avoid duplicates.
        """
        code_files = context.get("code_files", [])
        execution_logs = context.get("execution_logs", [])
        existing_skills = context.get("existing_skills", [])
        return {
            "code_files": code_files,
            "execution_logs": execution_logs,
            "existing_skills": existing_skills,
            "project_type": context.get("project_type", "unknown"),
        }

    async def reason(self, perception: dict[str, Any]) -> dict[str, Any]:
        """Identify recurring code patterns using LLM analysis.

        Uses LLM to analyze code files for:
        1. Authentication/authorization patterns
        2. CRUD operation templates
        3. Form validation patterns
        4. API client patterns
        5. Error handling patterns
        6. Testing patterns

        Filters out patterns that already exist in the skill registry.
        """
        # LLM call would go here to analyze code_files and identify patterns
        # For now, return the perception for the act phase to process
        return {
            "patterns_to_extract": perception["code_files"],
            "existing_skills": perception["existing_skills"],
        }

    async def act(self, plan: dict[str, Any]) -> AgentResult:
        """Create skills from identified patterns.

        For each pattern:
        1. Create Skill object with input/output schemas
        2. Register via SkillService (starts as DRAFT)
        3. Run validation tests
        4. If tests pass, activate the skill
        5. Emit skill.created event
        """
        created_skills: list[str] = []
        errors: list[str] = []

        patterns: list[ExtractedPattern] = plan.get("extracted_patterns", [])
        for pattern in patterns:
            try:
                skill = Skill(
                    id=uuid.uuid4(),
                    name=pattern.name,
                    description=pattern.description,
                    version="1.0.0",
                    created_by_agent="skill_creator",
                    target_agents=pattern.applicable_agents,
                    code=pattern.parameterized_code,
                    input_schema=pattern.input_schema,
                    output_schema=pattern.output_schema,
                    tags=pattern.tags,
                )
                created = await self._skill_service.create_skill(skill)
                # In production, run tests here before activating
                await self._skill_service.activate_skill(created.id)
                await self._event_bus.publish("skill.created", {
                    "skill_id": str(created.id),
                    "name": created.name,
                    "version": created.version,
                    "created_by": "skill_creator",
                })
                created_skills.append(str(created.id))
            except Exception as e:
                errors.append(f"Failed to create skill '{pattern.name}': {e}")

        return AgentResult(
            success=len(errors) == 0,
            output={"created_skills": created_skills, "errors": errors},
            error="; ".join(errors) if errors else None,
        )
```

```python
from __future__ import annotations
import uuid
from dataclasses import dataclass, field
from typing import Any
from codebot.agents.base import BaseAgent, AgentResult
from codebot.hooks.service import HookService
from codebot.events.bus import EventBus

@dataclass(slots=True, kw_only=True)
class HookDefinition:
    name: str
    hook_type: str  # maps to HookType enum values
    target: str
    priority: int = 100
    description: str = ""
    implementation: str = ""  # async callable source code

class HooksCreatorAgent(BaseAgent):
    """Agent #27: Generates lifecycle hooks for pipeline customization.

    Runs post-delivery in Stage S9. Analyzes the project's pipeline configuration
    and execution patterns to propose and register lifecycle hooks that improve
    pipeline behavior (pre/post phase hooks, error handlers, approval hooks).
    """

    def __init__(self, config: Any, hook_service: HookService, event_bus: EventBus) -> None:
        super().__init__(config)
        self._hook_service = hook_service
        self._event_bus = event_bus

    async def perceive(self, context: dict[str, Any]) -> dict[str, Any]:
        """Gather pipeline config, execution history, and existing hooks."""
        return {
            "pipeline_config": context.get("pipeline_config", {}),
            "execution_history": context.get("execution_history", []),
            "existing_hooks": context.get("existing_hooks", []),
            "project_type": context.get("project_type", "unknown"),
        }

    async def reason(self, perception: dict[str, Any]) -> dict[str, Any]:
        """Propose lifecycle hooks based on pipeline analysis.

        Identifies opportunities for:
        1. Pre-phase validation hooks (input checks before expensive stages)
        2. Post-phase metric collection hooks
        3. Error recovery hooks for common failure patterns
        4. Notification hooks for human approval gates
        5. Logging/telemetry hooks for observability
        """
        return {"proposed_hooks": [], "perception": perception}

    async def act(self, plan: dict[str, Any]) -> AgentResult:
        """Register proposed hooks via HookService."""
        from codebot.hooks.models import Hook, HookType, HookStatus

        registered: list[str] = []
        errors: list[str] = []

        definitions: list[HookDefinition] = plan.get("proposed_hooks", [])
        for defn in definitions:
            try:
                hook = Hook(
                    id=str(uuid.uuid4()),
                    name=defn.name,
                    hook_type=HookType(defn.hook_type),
                    target=defn.target,
                    priority=defn.priority,
                    status=HookStatus.PENDING,
                )
                await self._hook_service.register(hook)
                await self._event_bus.publish("hook.created", {
                    "hook_id": hook.id,
                    "name": hook.name,
                    "type": defn.hook_type,
                    "target": defn.target,
                })
                registered.append(hook.id)
            except Exception as e:
                errors.append(f"Failed to register hook '{defn.name}': {e}")

        return AgentResult(
            success=len(errors) == 0,
            output={"registered_hooks": registered, "errors": errors},
            error="; ".join(errors) if errors else None,
        )
```

```python
from __future__ import annotations
import json
from dataclasses import dataclass, field
from typing import Any
from codebot.agents.base import BaseAgent, AgentResult
from codebot.tools.service import ToolService
from codebot.events.bus import EventBus

@dataclass(slots=True, kw_only=True)
class ToolSpec:
    name: str
    description: str
    parameters: dict[str, Any]  # JSON Schema
    implementation: str  # Python source for the async execute function
    version: str = "1.0.0"
    tags: list[str] = field(default_factory=list)

class ToolsCreatorAgent(BaseAgent):
    """Agent #28: Generates custom tools and MCP server configurations.

    Runs post-delivery in Stage S9. Accepts natural-language tool descriptions,
    designs parameter schemas, generates async implementations, registers tools
    in the ToolService, and optionally scaffolds MCP server configuration for
    external tool exposure.
    """

    def __init__(self, config: Any, tool_service: ToolService, event_bus: EventBus) -> None:
        super().__init__(config)
        self._tool_service = tool_service
        self._event_bus = event_bus

    async def perceive(self, context: dict[str, Any]) -> dict[str, Any]:
        """Gather tool requests and existing registry."""
        return {
            "tool_requests": context.get("tool_requests", []),
            "existing_tools": context.get("existing_tools", []),
            "project_context": context.get("project_context", {}),
        }

    async def reason(self, perception: dict[str, Any]) -> dict[str, Any]:
        """Design tool interfaces using LLM.

        For each tool request:
        1. Design the parameter JSON Schema
        2. Determine appropriate tags and version
        3. Plan the implementation approach
        4. Check for duplicates against existing_tools
        """
        return {"tool_specs": [], "perception": perception}

    async def act(self, plan: dict[str, Any]) -> AgentResult:
        """Generate tool implementations and register them.

        For each tool spec:
        1. Validate parameter schema is valid JSON Schema
        2. Generate async execute function implementation
        3. Register via ToolService.create_tool()
        4. Generate MCP server config entry
        5. Emit tool.created event
        """
        created_tools: list[str] = []
        mcp_configs: list[dict[str, Any]] = []
        errors: list[str] = []

        specs: list[ToolSpec] = plan.get("tool_specs", [])
        for spec in specs:
            try:
                # Validate schema structure
                if not isinstance(spec.parameters, dict) or "type" not in spec.parameters:
                    raise ValueError(f"Invalid parameter schema for tool '{spec.name}': must be a JSON Schema object with 'type' field")

                # Create a minimal ToolDefinition for registration
                # In production, the implementation string would be compiled into an async callable
                from codebot.tools.registry import ToolDefinition

                async def _placeholder_execute(**kwargs: Any) -> str:
                    return f"Tool {spec.name} executed with {kwargs}"

                definition = ToolDefinition(
                    name=spec.name,
                    description=spec.description,
                    parameters=spec.parameters,
                    execute=_placeholder_execute,
                    version=spec.version,
                    tags=spec.tags,
                )
                await self._tool_service.create_tool(definition)

                # Generate MCP config entry
                mcp_entry = {
                    "name": spec.name,
                    "description": spec.description,
                    "inputSchema": spec.parameters,
                }
                mcp_configs.append(mcp_entry)

                await self._event_bus.publish("tool.created", {
                    "name": spec.name,
                    "version": spec.version,
                    "created_by": "tools_creator",
                })
                created_tools.append(spec.name)
            except Exception as e:
                errors.append(f"Failed to create tool '{spec.name}': {e}")

        # Generate aggregate MCP server config
        mcp_server_config = {
            "mcpServers": {
                "codebot-custom-tools": {
                    "command": "python",
                    "args": ["-m", "codebot.tools.mcp_server"],
                    "tools": mcp_configs,
                }
            }
        }

        return AgentResult(
            success=len(errors) == 0,
            output={
                "created_tools": created_tools,
                "mcp_config": mcp_server_config,
                "errors": errors,
            },
            error="; ".join(errors) if errors else None,
        )

    def generate_mcp_config(self, tools: list[ToolSpec]) -> dict[str, Any]:
        """Generate MCP server configuration JSON for the created tools."""
        return {
            "mcpServers": {
                "codebot-custom-tools": {
                    "command": "python",
                    "args": ["-m", "codebot.tools.mcp_server"],
                    "tools": [
                        {"name": t.name, "description": t.description, "inputSchema": t.parameters}
                        for t in tools
                    ],
                }
            }
        }
```
