   ```python
   class PipelineEventEmitter:
       """Wraps EventBus to emit structured pipeline and agent events.

       Usage:
           emitter = PipelineEventEmitter(bus=event_bus, pipeline_id=uuid)
           await emitter.agent_started(agent_type=AgentType.ORCHESTRATOR, agent_id=uuid)
           await emitter.agent_completed(agent_type=AgentType.ORCHESTRATOR, agent_id=uuid)
           await emitter.phase_started(phase_name="input_processing")
           await emitter.phase_completed(phase_name="input_processing")
           await emitter.pipeline_started()
           await emitter.pipeline_completed()
       """

       def __init__(self, bus: EventBus, pipeline_id: uuid.UUID) -> None:
           self._bus = bus
           self._pipeline_id = pipeline_id

       async def agent_started(self, agent_type: AgentType, agent_id: uuid.UUID) -> None:
           """Emit AGENT_STARTED event."""
           event = AgentEvent(agent_id=agent_id, agent_type=agent_type, status=AgentStatus.RUNNING)
           envelope = EventEnvelope(
               event_type=EventType.AGENT_STARTED,
               source_agent_id=agent_id,
               payload=event.model_dump_json().encode(),
           )
           await publish_event(self._bus, envelope)

       async def agent_completed(self, agent_type: AgentType, agent_id: uuid.UUID) -> None:
           """Emit AGENT_COMPLETED event."""
           # Similar to agent_started but with AgentStatus.COMPLETED and EventType.AGENT_COMPLETED

       async def agent_failed(self, agent_type: AgentType, agent_id: uuid.UUID, error: str = "") -> None:
           """Emit AGENT_FAILED event."""
           # AgentStatus.FAILED, EventType.AGENT_FAILED, include error in payload

       async def phase_started(self, phase_name: str) -> None:
           """Emit PHASE_STARTED event."""
           event = PipelineEvent(pipeline_id=self._pipeline_id, phase=phase_name, status="RUNNING")
           envelope = EventEnvelope(
               event_type=EventType.PHASE_STARTED,
               payload=event.model_dump_json().encode(),
           )
           await publish_event(self._bus, envelope)

       async def phase_completed(self, phase_name: str) -> None:
           """Emit PHASE_COMPLETED event."""

       async def pipeline_started(self) -> None:
           """Emit PIPELINE_STARTED event."""

       async def pipeline_completed(self) -> None:
           """Emit PIPELINE_COMPLETED event."""
   ```

   ```python
           Flow:
           1. PHASE: input_processing
              - Emit phase_started
              - OrchestratorAgent: perceive -> reason -> act -> review
              - Emit agent_started/completed for orchestrator
              - Store requirements in shared_state
              - Emit phase_completed

           2. PHASE: implementation
              - Emit phase_started
              - BackendDevAgent: perceive -> reason -> act -> review
              - Emit agent_started/completed for backend_dev
              - Store generated files in shared_state
              - Emit phase_completed

              - Emit phase_started
              - CodeReviewerAgent: perceive -> reason -> act -> review
              - If gate_passed=False: reroute to implementation (max 2 cycles)
              - Emit agent_started/completed for code_reviewer
              - Emit phase_completed

           4. PHASE: testing
              - Emit phase_started
              - TesterAgent: perceive -> reason -> act -> review
              - Store test results in shared_state
              - Emit agent_started/completed for tester
              - Emit phase_completed

              - If tests_passing=True: skip
              - Emit phase_started
              - DebuggerAgent: perceive -> reason -> act -> review
              - Emit agent_started/completed for debugger
              - Emit phase_completed

           Returns final shared_state.
           """
   ```
