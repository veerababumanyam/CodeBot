    ```python
    from codebot.pipeline.models import PipelineConfig

    def load_preset(
        preset_name: str,
        config_dir: Path | None = None,
    ) -> PipelineConfig:
        if config_dir is None:
            config_dir = Path(__file__).resolve().parents[5] / "configs" / "pipelines"
        path = config_dir / f"{preset_name}.yaml"
        if not path.exists():
            raise FileNotFoundError(f"Pipeline preset not found: {path}")
        with open(path) as f:
            raw = yaml.safe_load(f)
        return PipelineConfig.model_validate(raw["pipeline"])
    ```

    ```python
    from __future__ import annotations

    from pathlib import Path

    from agent_sdk.models.enums import ProjectType
    from codebot.pipeline.models import PipelineConfig


    def detect_project_type(
        repository_path: str,
        prd_content: str = "",
    ) -> ProjectType:
        """Detect whether a project is greenfield, inflight, or brownfield.

        Args:
            repository_path: Path to the project's git repository (empty string = no repo).
            prd_content: PRD text to scan for project type hints.

        Returns:
            ProjectType enum value.

        Detection heuristics:
        - No repository_path or empty string -> GREENFIELD
        - repository_path exists, has .git dir, has >50 source files -> BROWNFIELD
        - repository_path exists, has .git dir, has source files but <50 -> INFLIGHT
        - Explicit type hint in prd_content overrides heuristics
        """
        # Check for explicit type hints in PRD
        prd_lower = prd_content.lower()
        if "project_type: brownfield" in prd_lower or "brownfield" in prd_lower.split("project type:")[-1][:50] if "project type:" in prd_lower else False:
            return ProjectType.BROWNFIELD
        if "project_type: inflight" in prd_lower:
            return ProjectType.INFLIGHT

        # No repo path -> new project
        if not repository_path:
            return ProjectType.GREENFIELD

        repo = Path(repository_path)
        if not repo.exists() or not (repo / ".git").exists():
            return ProjectType.GREENFIELD

        # Count source files (py, ts, js, go, rs, java, etc.)
        source_extensions = {".py", ".ts", ".tsx", ".js", ".jsx", ".go", ".rs", ".java", ".kt", ".swift", ".rb"}
        source_count = sum(
            1 for f in repo.rglob("*")
            if f.suffix in source_extensions and "node_modules" not in str(f) and ".git" not in str(f)
        )

        if source_count > 50:
            return ProjectType.BROWNFIELD
        if source_count > 0:
            return ProjectType.INFLIGHT

        return ProjectType.GREENFIELD


    def adapt_pipeline_for_project_type(
        config: PipelineConfig,
        project_type: ProjectType,
    ) -> PipelineConfig:
        """Filter pipeline phases based on project type.

        Phases with the project type's value (lowercased) in their
        skip_for_project_types list are removed.

        Args:
            config: The full pipeline configuration.
            project_type: Detected or specified project type.

        Returns:
            A new PipelineConfig with filtered phases.
        """
        type_value = project_type.value.lower()
        adapted_phases = [
            phase for phase in config.phases
            if type_value not in [t.lower() for t in phase.skip_for_project_types]
        ]
        return config.model_copy(update={"phases": adapted_phases})
    ```
