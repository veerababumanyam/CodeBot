    from __future__ import annotations

    import json
    from dataclasses import dataclass, field
    from datetime import datetime, timezone

    import nats
    from nats.js.api import StreamConfig


    @dataclass
    class PipelineEvent:
        type: str
        data: dict = field(default_factory=dict)
        timestamp: str = field(
            default_factory=lambda: datetime.now(timezone.utc).isoformat()
        )

        def to_json_bytes(self) -> bytes:
            payload = {
                "type": self.type,
                "timestamp": self.timestamp,
                **self.data,
            }
            return json.dumps(payload).encode()

        @property
        def subject(self) -> str:
            return f"pipeline.{self.type.replace('.', '_')}"


    class PipelineEventEmitter:
        """Publishes pipeline events to NATS JetStream.

        Events are published to subjects matching "pipeline.{event_type}".
        A JetStream stream named PIPELINE_EVENTS captures all pipeline.*
        subjects with 7-day retention.
        """

        STREAM_NAME = "PIPELINE_EVENTS"

        def __init__(self, nc: nats.NATS) -> None:
            self._nc = nc
            self._js = nc.jetstream()

        async def ensure_stream(self) -> None:
            """Create or update the JetStream stream for pipeline events."""
            await self._js.add_stream(
                StreamConfig(
                    name=self.STREAM_NAME,
                    subjects=["pipeline.>"],
                    retention="limits",
                    max_age=7 * 24 * 3600 * 1_000_000_000,  # 7 days in nanoseconds
                )
            )

        async def emit(self, event_type: str, data: dict | None = None) -> None:
            """Emit a pipeline event to JetStream."""
            event = PipelineEvent(type=event_type, data=data or {})
            await self._js.publish(event.subject, event.to_json_bytes())

        async def emit_phase_started(
            self, phase_name: str, phase_idx: int, project_id: str = ""
        ) -> None:
            await self.emit("phase.started", {
                "phase": phase_name,
                "phase_idx": phase_idx,
                "project_id": project_id,
            })

        async def emit_phase_completed(
            self, phase_name: str, phase_idx: int, project_id: str = ""
        ) -> None:
            await self.emit("phase.completed", {
                "phase": phase_name,
                "phase_idx": phase_idx,
                "project_id": project_id,
            })

        async def emit_gate_waiting(self, gate_id: str) -> None:
            await self.emit("gate.waiting", {"gate_id": gate_id})

        async def emit_gate_decided(
            self, gate_id: str, decision: str
        ) -> None:
            await self.emit("gate.decided", {
                "gate_id": gate_id,
                "decision": decision,
            })
    ```

    import logging
    import nats as nats_lib
    from codebot.pipeline.events import PipelineEventEmitter

    logger = logging.getLogger(__name__)

    # Module-level emitter singleton (set by worker on startup)
    _emitter: PipelineEventEmitter | None = None


    def set_event_emitter(emitter: PipelineEventEmitter) -> None:
        """Set the shared event emitter (called by worker on startup)."""
        global _emitter
        _emitter = emitter


    @activity.defn
    async def emit_pipeline_event(event_data: dict) -> None:
        """Emit a pipeline event to NATS JetStream.

        Uses module-level singleton emitter set by the worker.
        Falls back to logging if NATS is unavailable.
        """
        event_type = event_data.get("type", "unknown")
        activity.heartbeat(f"Emitting event: {event_type}")

        if _emitter is not None:
            try:
                await _emitter.emit(event_type, event_data)
                return
            except Exception as exc:
                logger.warning("NATS emit failed, falling back to log: %s", exc)

        # Fallback: log the event
        timestamp = datetime.now(timezone.utc).isoformat()
        logger.info("Pipeline event: type=%s timestamp=%s data=%s", event_type, timestamp, event_data)
    ```
